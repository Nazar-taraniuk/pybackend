# Лаба 2 — Послідовність команд
# Docker + FastAPI + PostgreSQL

## 1. Додавання залежностей через Poetry

# Додаємо asyncpg (async драйвер PostgreSQL) та SQLAlchemy (ORM)
poetry add asyncpg sqlalchemy


## 2. Збірка і запуск контейнерів

# Збудувати образи і підняти контейнери у фоні (-d = detached)
docker compose up --build -d

# Якщо потрібно перезапустити з нуля (видалити volumes)
docker compose down -v
docker compose up --build -d


## 3. Перевірка стану контейнерів

# Список запущених контейнерів з портами і статусом
docker compose ps

# Логи FastAPI сервісу (останні 20 рядків)
docker compose logs api --tail=20

# Логи PostgreSQL сервісу
docker compose logs db --tail=20


## 4. Список бібліотек всередині контейнера

# Заходимо всередину контейнера і дивимось встановлені пакети
docker exec pybackend_api pip list


## 5. Перевірка роботи API (у браузері або curl)

# Swagger документація
# http://localhost:8000/docs

# Перевірка підключення до PostgreSQL
# http://localhost:8000/db-check

# Health check
# http://localhost:8000/health


## 6. Автоперезавантаження при зміні коду

# Завдяки volume mount (. :/app) і прапорцю --reload у uvicorn
# сервер автоматично перезапускається при зміні будь-якого .py файлу
# Перевірка: зміни app/main.py і дивись логи:
docker compose logs api -f


## 7. Зупинка контейнерів

# Зупинити без видалення
docker compose stop

# Зупинити і видалити контейнери (але зберегти volume з даними БД)
docker compose down

# Зупинити і видалити все включно з volumes (дані БД будуть втрачені)
docker compose down -v

# Лаба 3 — CRUD для юзерів (FastAPI + Pydantic + Роутери)

## 1. Встановлення нової залежності (валідація email)

# pydantic[email] = pydantic + email-validator
poetry add "pydantic[email]"


## 2. Структура проекту після лаби 3

# d:\pybackend\
# ├── main.py                    ← точка входу (запускає uvicorn)
# ├── Dockerfile                 ← оновлено: запуск через python main.py
# ├── app/
# │   ├── main.py               ← FastAPI app + підключення роутерів
# │   ├── routers/
# │   │   └── users.py          ← CRUD роутер для юзерів
# │   └── schemas/
# │       └── user.py           ← Pydantic схеми валідації


## 3. Перебудова контейнера після змін

docker compose down
docker compose up --build -d

# ============================================================
# Лаба 4 — PostgreSQL + Alembic + Pydantic Settings + CRUD
# ============================================================

## 1. Встановлення нових залежностей

poetry add alembic pydantic-settings

## 2. Ініціалізація Alembic

poetry run alembic init alembic

## 3. Генерація першої міграції (запускати з середини контейнера або через docker run)

# Генерація автоміграції на основі моделей
docker exec pybackend_api alembic revision --autogenerate -m "initial_tables"

# Або через тимчасовий контейнер (якщо api ще не запущено):
docker run --rm --network pybackend_default -v "${PWD}:/app" pybackend-api alembic revision --autogenerate -m "initial_tables"

## 4. Застосування міграцій та заповнення даними

# Міграції + seed відбуваються автоматично при старті контейнера (CMD в Dockerfile)
docker compose up --build -d

# Або вручну всередині контейнера:
docker exec pybackend_api alembic upgrade head
docker exec pybackend_api python seed.py

## 5. Перевірка даних у БД через psql

docker exec pybackend_db psql -U admin -d pybackend -c "\dt"
docker exec pybackend_db psql -U admin -d pybackend -c "SELECT * FROM categories;"
docker exec pybackend_db psql -U admin -d pybackend -c "SELECT * FROM users;"
docker exec pybackend_db psql -U admin -d pybackend -c "SELECT * FROM products;"


# ============================================================
# Лаба 5 — JWT Аутентифікація + HTTP-only Cookie + bcrypt
# ============================================================

## 1. Встановлення залежностей

poetry add "python-jose[cryptography]" bcrypt

## 2. Нова міграція для поля password_hash

docker run --rm --network pybackend_default -v "${PWD}:/app" pybackend-api alembic revision --autogenerate -m "add_password_hash_to_users"

## 3. API ендпоінти аутентифікації

# Реєстрація нового юзера
# POST http://localhost:8000/auth/register
# Body: {"name": "Test User", "email": "test@example.com", "password": "password123"}

# Вхід (JWT записується в HTTP-only куку)
# POST http://localhost:8000/auth/login
# Body: {"email": "test@example.com", "password": "password123"}

# Поточний юзер (захищений endpoint, потребує куки)
# GET http://localhost:8000/auth/me

# Мої замовлення (захищений endpoint)
# GET http://localhost:8000/auth/my-orders

# Вихід (очищення куки)
# POST http://localhost:8000/auth/logout
